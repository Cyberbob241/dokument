\textbf{{\large Die 2  (Neubiberg)}}

\begin{enumerate}
	\item Fragen zur Blockchain (Wissensstand: Was sind Kryptowährungen? Was ist die Blockchain? Anwendungsgebiete?)
	\begin{itemize}
		\item Kryptowährungen meist eine eigene Blockchain, funktioniert dann ein virtueller "Ledger", also eine Art Bestandsbuch, in dem jede Transaktion gespeichert wird
		\item hervorragend als dezentrale Währung
		\item direkte Verbindung zwischen allen Teilnehmern des Netzwerkes aufgebaut und somit ist es einer einzelnen Institution nicht möglich, den Geldfluss zu korrumpieren
		\item Anwendungsgebiete besonders beim "Digital Cash" und Micro-Transaktionen
		\item Durch Blockchain kann Geld relativ schnell, zügig und sicher  Benutzer wechseln
	\end{itemize}
	\item Erklärung der Blockchain, Erklärung von Ethereum 
	\begin{itemize}
		\item entfällt, da umfangreiches Wissen vorhanden
	\end{itemize}
	\item Fragen über die derzeitige Situation der Bar (Bestellvorgang, Bezahlung, Webauftritt)
	\begin{itemize}
		\item normaler Bestellvorgang 
		\item möglich mit EC-Karte und Bargeld zu bezahlen
		\item mehr Kunden nutzten die Kartenzahlung, da Bargeld öfter vergessen
		\item umständlicher als mit Bargeld, da der Bezahlvorgang mit dem EC-Kartengerät wesentlich länger dauert
		\item eigener Webauftritt
	\end{itemize}
	\item Vorstellung des Szenarios
	\begin{itemize}
		\item Kurze Erklärung des Szenarios
		\item Erläuterung der Vor- und Nachteile
	\end{itemize} 
	\item Fragen über das Szenario: Vorstellbar, Heute Umsetzbar (Begründung für ja und nein)
	\begin{itemize}
		\item viele Menschen vertrauen dem Blockchain-Prinzip nicht (Kursschwankungen)
		\item Marketing Technisch könnte es natürlich vorteilhaft sein, wenn man zu den "ersten" gehört
		\item Art Alleinstellungsmerkmal - "Cool"
		\item Wenn Kosten für Anschaffung nicht zu hoch sind, hört sich das auf jeden Fall interessant an
		
	\end{itemize}
	\item Abschließende Worte
	\begin{itemize}
		\item definitiv nicht, dass die Blockchaintechnologie in naher Zukunft das Bargeld ablöst. 
		\item Bargeld auch verwenden wenn der Akku mal leer ist oder man kein Netz hat
	\end{itemize}
\end{enumerate}

\textbf{{\large Unicasino  (Neubiberg)}}

\begin{enumerate}
	\item Fragen zur Blockchain (Wissensstand: Was sind Kryptowährungen? Was ist die Blockchain? Anwendungsgebiete?)
	\begin{itemize}
		\item durch Whitepaper Bitcoin/Blockchain und Ethereum und News einigermaßen 'tiefgehendes' Verständnis der Technologie/technischen Komponenten 
		\item Blockchain: Kohärente, Konsistenz- und Integritätsichernde Transaktionstechnologie
		\item Kryptowährungen: Basierend auf Blockchain dezentrale, partiell anonymisierbare Transaktionen  möglich; 
		\item Verzicht auf zentrale Vertrauensinstanz (wie beispielsweise einer Bank) 
		\item soll Regulierung und Kontrolle erschweren
		
	\end{itemize}
	\item Erklärung der Blockchain, Erklärung von Ethereum 
	\begin{itemize}
		\item Kurze Erklärung über die Technologie und Anwendungsmöglichkeiten
	\end{itemize}
	\item Fragen über die derzeitige Situation der Bar (Bestellvorgang, Bezahlung, Webauftritt)
	\begin{itemize}
		\item Bezahlung über Bargeld oder EC-Karte
		\item Unterschiedliche Bestellungs-/Bezahlvorgänge (Restaurant und Bar)
		\begin{itemize}
			\item Bar erst bezahlen, dann Getränk
			\item Restaurant erst Essen/Trinken, am Ende bezahlen (Trinkgeldaspekt)
		\end{itemize}
		\item Bestellvorgang manuell, Bargeld eventuell hinderlich/unhandlich
		\item Website für allgemeine Informationen, Werbung für Veranstaltungen und Onlinereservierungen
		
	\end{itemize}
	\item Vorstellung des Szenarios
	\begin{itemize}
		\item Kurze Erklärung des Szenarios
		\item Erläuterung der Vor- und Nachteile
	\end{itemize} 
	\item Fragen über das Szenario: Vorstellbar, Heute Umsetzbar (Begründung für ja und nein)
	\begin{itemize}
		\item Da Mobilgerät immer dabei, Bezahloption naheliegend
		\item Mit Wallet auf Handy Transaktionen einfach zu realisieren
		\item Gerade im aktuellen Blockchain-Hype Potential nutzbar ('Hipster'-Bar)
		\item Webauftritt heute eigentlich obligatorisch, Integration Bezahlfunktionalität sinnvoll
		\item Feinheiten der Umsetzung schwierig (Wie findet erste Authentifizierung des Nutzers statt?)
		\item Umsetzbarkeit wohl nur aus marketing-technischer Sicht sinnvoll, Konzept bringt weder Betreiber noch Endkunde signifikante Komfort-Vorteil
		\item Umsetzung nur als Trend-Konzept, Technologie liefert keinen produktiver Beitrag 
		
		
	\end{itemize}
	\item Abschließende Worte
	\begin{itemize}
		\item Blockchain-Technologie in der praktischen Verwendbarkeit stark limitiert
		\item Insbesondere Skalierbarkeit und Stabilität der Währungs-Anteile kritisch
	\end{itemize}
\end{enumerate}
